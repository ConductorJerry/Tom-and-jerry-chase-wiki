### Code with Explanation

The following function demonstrates lorem ipsum generation:

```typescript
interface LoremIpsumConfig {
  wordCount: number;
  startWithLorem: boolean;
}

function generateLoremIpsum(config: LoremIpsumConfig): string {
  const words = [
    "lorem", "ipsum", "dolor", "sit", "amet", "consectetur",
    "adipiscing", "elit", "sed", "do", "eiusmod", "tempor"
  ];
  
  // Generate random words
  const result = [];
  if (config.startWithLorem) {
    result.push("Lorem", "ipsum");
  }
  
  for (let i = result.length; i < config.wordCount; i++) {
    const randomIndex = Math.floor(Math.random() * words.length);
    result.push(words[randomIndex]);
  }
  
  return result.join(" ") + ".";
}
```
